from selenium import webdriver

        driver.switch_to.window(driver.window_handles[0])

## **Q45 ‚Äî Take screenshots automatically on test failure (pytest fixture example)**

File: tests/test_with_screenshot.py ‚Äî run with pytest --html=report.html

```
# tests/test_with_screenshot.py
import pytest
from selenium import webdriver
from selenium.webdriver.chrome.service import Service
from webdriver_manager.chrome import ChromeDriverManager
import os

@pytest.fixture
def driver():
    driver = webdriver.Chrome(service=Service(ChromeDriverManager().install()))
    driver.maximize_window()
    yield driver
    driver.quit()

@pytest.hookimpl(hookwrapper=True)
def pytest_runtest_makereport(item, call):
    # Hook to take screenshot on failure
    outcome = yield
    rep = outcome.get_result()
    if rep.when == "call" and rep.failed:
        driver = item.funcargs.get("driver")
        if driver:
            os.makedirs("failure_screenshots", exist_ok=True)
            path = f"failure_screenshots/{item.name}.png"
            driver.save_screenshot(path)
            print(f"Saved failure screenshot: {path}")

def test_example_fail(driver):
    driver.get("https://demoqa.com/")
    assert False, "Intentional failure to demo screenshot"
```

## **Q46 ‚Äî Script to capture broken links from any site and log to a file**

```
# q46_broken_links.py
import requests
from selenium import webdriver
from selenium.webdriver.common.by import By
from selenium.webdriver.chrome.service import Service
from webdriver_manager.chrome import ChromeDriverManager
import time

driver = webdriver.Chrome(service=Service(ChromeDriverManager().install()))
driver.maximize_window()
site = "https://demoqa.com"
driver.get(site)
time.sleep(2)

links = driver.find_elements(By.TAG_NAME, "a")
broken = []

for a in links:
    href = a.get_attribute("href")
    if href and href.startswith("http"):
        try:
            r = requests.head(href, allow_redirects=True, timeout=5)
            if r.status_code >= 400:
                broken.append((href, r.status_code))
                print("Broken:", href, r.status_code)
        except Exception as e:
            broken.append((href, str(e)))
            print("Error checking:", href, e)

with open("broken_links.log", "w") as f:
    for u, status in broken:
        f.write(f"{u} -> {status}\n")

print("Done. Broken links logged to broken_links.log")
driver.quit()
```

from selenium import webdriver
from selenium.webdriver.common.by import By
from selenium.webdriver.chrome.service import Service as ChromeService
from selenium.webdriver.chrome.options import Options
from webdriver_manager.chrome import ChromeDriverManager
from datetime import datetime
import time
import requests

# ---------- Setup ----------
options = Options()
options.add_argument("--headless")  # Run without opening browser window
driver = webdriver.Chrome(service=ChromeService(ChromeDriverManager().install()), options=options)

report_lines = []

try:
    start_time = datetime.now().strftime("%Y-%m-%d %H:%M:%S")
    report_lines.append(f"üßæ IMCC Website Automation Report ({start_time})\n")

    # 1Ô∏è‚É£ Open site
    driver.get("https://imcc.edu.in/")
    time.sleep(3)
    report_lines.append("‚úÖ Website opened successfully.\n")

    # 2Ô∏è‚É£ Validate title
    title = driver.title
    expected = "IMCC"
    if expected.lower() in title.lower():
        report_lines.append(f"‚úÖ Title verified: {title}\n")
    else:
        report_lines.append(f"‚ùå Title mismatch: {title}\n")

    # 3Ô∏è‚É£ Check all image URLs
    images = driver.find_elements(By.TAG_NAME, "img")
    report_lines.append(f"üîç Found {len(images)} images. Checking for broken links...\n")

    broken_count = 0
    for img in images:
        src = img.get_attribute("src")
        try:
            res = requests.head(src, timeout=5)
            if res.status_code != 200:
                report_lines.append(f"‚ùå Broken Image: {src}\n")
                broken_count += 1
        except:
            broken_count += 1
            report_lines.append(f"‚ö†Ô∏è Unreachable Image: {src}\n")
    report_lines.append(f"üìä Total Broken Images: {broken_count}\n")

    # 4Ô∏è‚É£ Navigate all top menu links
    menus = driver.find_elements(By.CSS_SELECTOR, "ul.navbar-nav li a")
    report_lines.append(f"üîó Found {len(menus)} menu links. Navigating...\n")

    for menu in menus:
        link = menu.get_attribute("href")
        text = menu.text.strip()
        if link:
            driver.get(link)
            report_lines.append(f"‚úÖ Visited: {text} ({link})\n")
            time.sleep(1)

    # 5Ô∏è‚É£ Save report with timestamp
    timestamp = datetime.now().strftime("%Y%m%d_%H%M%S")
    report_file = f"imcc_test_report_{timestamp}.txt"
    with open(report_file, "w", encoding="utf-8") as f:
        f.writelines(report_lines)

    print(f"üìÅ Test report generated: {report_file}")

except Exception as e:
    print("‚ùå Error during test execution:", e)

finally:
    driver.quit()
